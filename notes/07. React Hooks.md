### useEffect()

只有 re-render 后，才执行 useEffect
- `[]` : 第一次 render 后立刻执行 useEffect, 之后再也不执行
- `[state]` : state 值改变引起的 re-render 后才执行 useEffect
- `漏写 []` : 任何地方 re-render 都执行 useEffect

```js
useEffect(()=>{}, [])
```

```js
const [state, setState] = useState(initialState);
//🔴直接用就行，不用 const
useEffect(() => {
	setState(newState);
}, [state]);
```

## Routes, Route / createBrowserRouter() / RouterProvider

```jsx
import { 
	Routes, Route, createBrowserRouter, RouterProvider
} from "react-router-dom";

import Header from "./component/Header";
import Nav from "./component/Nav";
import Footer from "./component/Footer";
import SubPage1 from "./main/SubPage1";
import SubPage2 from "./main/SubPage2";

function Root() {
	return(
		<>
			<Header />
			<Nav />
			<Routes>
				<Route path="/" element={<Home />} />
				<Route path="/SubPage1" element={<SubPage1 />} />
				<Route path="/SubPage2" element={<SubPage2 />} />
			</Routes>
			<Footer />
		</>	
	);
}

const router = createBrowserRouter(
	[{path:"*", Component:Root}]
);

export default function App() {
	return(
		<RouterProvider router={router} />
	);	
}
```

**注意：`creteBrowserRouter([{path:"*", element: <Root />}])`**

```jsx
import { Outlet, createBrowserRouter, RouterProvider } from "react-router-dom";

import 其他。。。

function Root() {
	return(
		<Header />
		<Nav />
		<main>
			<Outlet />  {/* 这里是子路由 */}
		</main>
		<Footer />
	);
}

const router = createBrowserRouter(
  [
    {   // 根路由
		path: "/",
		element: <Root />,  // 根布局
		
		// 子路由
		children: [
			{index: true, element: <Home />}, // 默认子路由
			{path: "profile", element: <Profile />}, // 子1
			{path: "contact", element: <Contact />}, // 子2
			{path: "course", element: <Course />}, // 子3
		],
    }
  ]
);

export default function App() {
	return(
		<RouterProvider router={router} />
	);
}
```

## document.getElementById() / ...ByClassName() / ...ByTagName()
- 写纯文本
- 获取数据
- 修改 DOM

```js
let input = getElementByID("id").value;  // 纯文本
let input = Number(input);  // 转换成数字
```

```html
<body>
	<div id="sum-calculator">
		<label for="first-input">Enter first number: </label>
		<input id="first-input" />
		
		<label for="second-input">Enter second number: </label>
		<input id="second-input" />
		
		<button onclick="Add()">Calculate Sum</button>
		
		<div id="output-div"></div>
	</div>

	<script>
		// 获取元素
		function Add() {
			const firstInput = Number(
				document.getElementById("first-input").value
			);
			const secondInput = Number(
				document.getElementById("second-input").value
			);
			const sum = firstInput + secondInput;
			
			// 修改 DOM
			document.getElementById("output-div").innerHTML = `
				<h1>Sum: ${sum}</h1>
			`;
		}
	</script>
</body>
```


## document.querySelector() / document.querySelectorAll()

- 获取第一个数据
- 获取所有数据

```js
// 函数里面写 "CSS selector 格式"
const firstTagInFooClass = document.querySelector(".foo");
const allTagInFooClass = document.querySelectorAll(".foo");
```
