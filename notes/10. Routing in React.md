# ✅【用这个】createBrowserRouter + RouterProvider

## App.jsx
- Root() --> Routes + Route
- router --> createBrowserRouter()
- App() --> RouterProvider

```jsx
/// 第一步: Root = Routes + Route
function Root() { 
	return (
		<>
			<Nav /> {/* 连接任何自制组件 */}
			<main>
				<Routes> {/* 这里是要跳转的部分 */}
					<Route path="/" element={<Page />} /> 
					<Route path="/foo" element={<Foo />} /> 
					...
				</Routes>
			</main>
			<Footer /> {/* 连接任何自制组件 */}
		</>
	);
}

/// 第二步: createBrowserRouter
const router = createBrowserRouter([{path:"*", Component:Root}]);

/// 第三步: export RouterProvider
export default function App() {
	return(
		<RouterProvider router={router} />
	);
}
```

> [!tip] 思考
> 1. 为什么用 `path:"*"` 而不是 `path:"/"` ? 效果有何不同？
> 2. 为什么用 `Component:Root` 而不是 `element:<Root />` ? 效果有何不同？
> 3. 如何用 `<Outlet />` 和 `children: [{path:..., element:...}]` 替代 `<Routes>` + `<Route>` ? 效果有何不同 ?

## App.jsx
用 Routes > `<Route path="/sub-page" element={<Child />} />` 绑定跳转路径

```jsx
// App.jsx
import { BrowserRouter, Routes, Route } from "react-router-dom";
import Navigation from "./components/Navigation";
import About from "./pages/About";
import Project from "./pages/Project";
import Contact from "./pages/Contact";

export default function App() {
  return (
    <BrowserRouter>
      <Navigation />
      <Routes>
        <Route path="/about" element={<About />} />
        <Route path="/project" element={<Project />} />
        <Route path="/contact" element={<Contact />} />
        <Route path="*" element={<h1>Page Not Found</h1>} />
      </Routes>
    </BrowserRouter>
  );
}
```

## CustomLink.jsx
用 useResolvedPath + useMatch + Link 判断当前路径是否是当前页面，是的话高光

```jsx
// CustomLink.jsx
import { useResolvedPath, useMatch, Link } from "react-router-dom";
import styled from "styled-components";

const ListItems = styled.li`
	margin-left: 0;
	color: ${(props) => props.className === "active" ? "red" : "blue"};
`

export default function CustomLink(props) {

	// resolvedPath()
	const resolvedPath = useResolvedPath(props.to);
	
	// useMatch()
	const isActive = useMatch({
		path: resolvedPath.pathname, 
		end: true
	});
		
	const linkStyles = {
		textDecoration: "none",
		fontSize: "calc(2px + 2vw)",
		color: "hotpink"
	};

	return (
		<ListItems className={isActive ? "active" : ""}>
			<Link to={props.to} style={linkStyles}>
				{props.children}
			</Link>
		</ListItems>
	)
}
```
